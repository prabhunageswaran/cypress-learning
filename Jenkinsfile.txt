pipeline {
	agent any
	stages {
		stage('Clone Git Repo'){
				steps{
					git 'https://github.com/prabhunageswaran/cypress-learning.git'
		    }
		}
		stage('Install Dependencies'){
				steps{
					bat 'npm install'
				}
		}
		stage('Run Tests'){
				steps{
					bat 'npm run dashBoard'
				}
		}
		stage('Publish HTML Report'){
				steps{
					publishHTML([allowMissing: false, alwaysLinkToLastBuild: true, keepAll: false, reportDir: 'mochawesome-report', reportFiles: 'summaryreport.html', reportName: 'SUMMARYREPORT', reportTitles: ''])
				}
		}
	}
}
